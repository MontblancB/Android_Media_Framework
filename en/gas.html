1 GAS Overview

Google Automotive Services (GAS) is a collection of Google applications and services that run on vehicles based on Android Automotive OS (AAOS).

Key Difference: GAS runs natively on the vehicle's built-in system, unlike Android Auto which projects from a smartphone.

üéØ What is GAS?

Google Automotive Services

  • Collection of Google apps running on AAOS
  • OEMs integrate into vehicles through licensing
  • Operates independently without smartphone
  • Direct integration with vehicle hardware

üì± vs Android Auto

  • Android Auto: Projects smartphone apps to vehicle display
  • GAS/AAOS: OS embedded in vehicle system
  • Android Auto requires smartphone
  • GAS is completely independent
  • GAS can integrate vehicle control features

üè¢ Major Manufacturers

Vehicle brands with GAS:

  • Volvo (including Polestar)
  • Renault
  • Ford
  • GM (Chevrolet, GMC, Cadillac)
  • Honda
  • BMW (select models)

‚ú® Key Benefits

  • No smartphone connection required
  • Integrated vehicle controls (AC, seats, etc.)
  • Continuous improvement via OTA updates
  • Familiar Google user experience
  • Google account synchronization

// MainActivity.kt - Using Location Services
class NavigationActivity : ComponentActivity() {
    private lateinit var fusedLocationClient: FusedLocationProviderClient

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        fusedLocationClient = LocationServices.getFusedLocationProviderClient(this)

        // Get current location
        if (ActivityCompat.checkSelfPermission(this,
            Manifest.permission.ACCESS_FINE_LOCATION) == PackageManager.PERMISSION_GRANTED) {

            fusedLocationClient.lastLocation.addOnSuccessListener { location ->
                location?.let {
                    Log.d("Location", "Lat: ${it.latitude}, Lng: ${it.longitude}")
                    updateMapPosition(it)
                }
            }
        }
    }

    // Real-time location updates
    private fun startLocationUpdates() {
        val locationRequest = LocationRequest.Builder(
            Priority.PRIORITY_HIGH_ACCURACY,
            10000 // Update every 10 seconds
        ).build()

        val locationCallback = object : LocationCallback() {
            override fun onLocationResult(result: LocationResult) {
                result.locations.forEach { location ->
                    updateNavigationRoute(location)
                }
            }
        }

        fusedLocationClient.requestLocationUpdates(
            locationRequest,
            locationCallback,
            Looper.getMainLooper()
        )
    }
}

            null  // addAccountOptions
        )
        startActivityForResult(intent, REQUEST_CODE_ADD_ACCOUNT)
    }
}

// Firebase Cloud Messaging Setup
class MyFirebaseMessagingService : FirebaseMessagingService() {

    // Receive new FCM token
    override fun onNewToken(token: String) {
        super.onNewToken(token)
        Log.d("FCM", "New token: $token")

        // Register token with server
        sendTokenToServer(token)
    }

    // Receive push message
    override fun onMessageReceived(remoteMessage: RemoteMessage) {
        super.onMessageReceived(remoteMessage)

        remoteMessage.notification?.let {
            // Display vehicle notification (minimize distraction while driving)
            if (isParked()) {
                showNotification(it.title, it.body)
            } else {
                // Voice-only notification while driving
                announceViaAssistant(it.body)
            }
        }

        // Handle data messages (OTA update notifications, etc.)
        remoteMessage.data.isNotEmpty().let {
            handleDataPayload(remoteMessage.data)
        }
    }

    private fun showNotification(title: String?, body: String?) {
        val notificationManager = getSystemService(Context.NOTIFICATION_SERVICE)
            as NotificationManager

        val notification = NotificationCompat.Builder(this, CHANNEL_ID)
            .setContentTitle(title)
            .setContentText(body)
            .setSmallIcon(R.drawable.ic_car)
            .setCategory(NotificationCompat.CATEGORY_CAR_INFORMATION)
            .build()

        notificationManager.notify(NOTIFICATION_ID, notification)
    }
}

üîÑ Google Play Services Version Check 

// Check Google Play Services availability
class BaseActivity : ComponentActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        checkGooglePlayServices()
    }

    private fun checkGooglePlayServices() {
        val apiAvailability = GoogleApiAvailability.getInstance()
        val resultCode = apiAvailability.isGooglePlayServicesAvailable(this)

        when (resultCode) {
            ConnectionResult.SUCCESS -> {
                Log.d("GMS", "Google Play Services available")
            }
            ConnectionResult.SERVICE_VERSION_UPDATE_REQUIRED -> {
                Log.w("GMS", "Google Play Services update required")
                // AAOS auto-updates, just inform user
                showUpdateRequiredDialog()
            }
            else -> {
                Log.e("GMS", "Google Play Services unavailable: $resultCode")
                // Critical error - most features unavailable without GAS
                showCriticalError()
            }
        }
    }
}

fun startNavigation(latitude: Double, longitude: Double, label: String) {
    val gmmIntentUri = Uri.parse("google.navigation:q=$latitude,$longitude")
    val mapIntent = Intent(Intent.ACTION_VIEW, gmmIntentUri)
    mapIntent.setPackage("com.google.android.apps.maps")

    if (mapIntent.resolveActivity(packageManager) != null) {
        startActivity(mapIntent)
    } else {
        // Google Maps not installed (AAOS without GAS)
        Toast.makeText(this, "Google Maps is not available", Toast.LENGTH_SHORT).show()
    }
}

        super.onCreate(savedInstanceState)

        handleAssistantIntent(intent)
    }

    override fun onNewIntent(intent: Intent?) {
        super.onNewIntent(intent)
        intent?.let { handleAssistantIntent(it) }
    }

    private fun handleAssistantIntent(intent: Intent) {
        when (intent.action) {
            Intent.ACTION_VIEW -> {
                val uri = intent.data
                if (uri?.scheme == "myapp" && uri.host == "play") {
                    val songName = uri.getQueryParameter("song")
                    playSong(songName)
                }
            }
        }
    }
}

                startActivity(intent)

sequenceDiagram participant Vehicle as Vehicle IVI participant GMS as Google Play Services participant PlayStore as Google Play Store participant Google as Google Update Server Note over Vehicle,Google: Automatic updates at night or when parked Vehicle->>GMS: Periodically check for updates GMS->>Google: Check for GAS app updates Google-->>GMS: Available updates list alt Updates available GMS->>PlayStore: Download update packages PlayStore->>Google: Download APK (Maps, Assistant, etc.) Google-->>PlayStore: APK files PlayStore->>PlayStore: Verify signature PlayStore->>Vehicle: Install updates (background) Vehicle->>Vehicle: Apply updates (A/B partition) Vehicle-->>GMS: Update complete alt Reboot required (System update) Vehicle->>Vehicle: Schedule reboot Note over Vehicle: Applied on next start end else No updates GMS-->>Vehicle: No updates available end

üîÑ A/B Partition System

üíæ Seamless Updates

How it works:

  • Partition A: Currently running system
  • Partition B: Update target
  • Update installed on B in background
  • Switch to B on reboot
  • Automatic rollback to A on failure

üì± GAS App Updates

Update frequency:

  • Google Maps: Monthly (including map data)
  • Google Assistant: 1-2 times monthly
  • Play Services: Monthly
  • YouTube Music: Every 2 weeks
  • Security patches: Immediate (when urgent)
